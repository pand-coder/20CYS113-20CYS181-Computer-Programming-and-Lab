 Pattern - * Full Pyramid

#include <stdio.h>

int main() {
    int r;
    scanf("%d", &r);
    if (r>0 && r<25)
    {
        for(int i = 0; i < r; i++) 
        {
            for(int j = 0; j < r-i-1; j++)
            {
                printf(" ");
            }
            for(int k=0; k<(2*i+1); k++)
            {
                printf("* ");
            }
            printf("\n");
        }
    }
    else {
        printf("Invalid Input");
    }
    return 0;
}
