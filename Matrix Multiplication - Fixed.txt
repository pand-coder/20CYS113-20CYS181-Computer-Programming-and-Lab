Matrix Multiplication - Fixed

#include<stdio.h>

int main()
{
    int A[3][3], B[3][3], C[3][3], i, j, k, sum = 0;
    
    for (i = 0; i < 3; i++)
    {
        for (j = 0; j < 3; j++) 
        {
            scanf("%d\t", &A[i][j]);
        }     
    }
    
    for (i = 0; i < 3; i++)
    {
        for (j = 0; j < 3; j++) 
        {
            scanf("%d\t", &B[i][j]);
        }     
    }
    
    for (i = 0; i <= 2; i++)
    {
        for (j = 0; j <= 2; j++) 
        {
            sum = 0;
            
            for (k = 0; k <= 2; k++)
            {
                sum = sum + A[i][k] * B[k][j];
            }
            
            C[i][j] = sum;
        }
    }
    
    for (i = 0; i < 3; i++)
    {
        for (j = 0; j < 3; j++)
        {
            printf(" %d ", C[i][j]);
        }
        
        printf("\n");
    }
    
    return 0;
}
